6#[derive(Copy, Clone, FromPrimitive, Eq, PartialEq, Debug, Ord, PartialOrd)]
7pub enum PieceType {
8    WhitePawn = 0,
9    WhiteRook = 1,
10    WhiteKnight = 2,
11    WhiteBishop = 3,
12    WhiteQueen = 4,
13    WhiteKing = 5,
14    BlackPawn = 6,
15    BlackRook = 7,
16    BlackKnight = 8,
17    BlackBishop = 9,
18    BlackQueen = 10,
19    BlackKing = 11,
20    Marker = 12,
21    SourceMarker = 13,
22    NoPiece,
23}
24
25impl fmt::Display for PieceType {
26    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
27        let pieces = [
28            "♟︎", "♜", "♞", "♝", "♛", "♚", "♙", "♖", "♘", "♗", "♕", "♔", "*", "+", " ",
29        ];
30        f.write_str(pieces[*self as usize])
31    }
32}

34impl PieceType {
35    pub fn is_black(self: &PieceType) -> bool {
36        matches!(
37            self,
38            PieceType::BlackPawn
39                | PieceType::BlackRook
40                | PieceType::BlackKnight
41                | PieceType::BlackBishop
42                | PieceType::BlackQueen
43                | PieceType::BlackKing
44        )
45    }
46
47    pub fn is_white(self: &PieceType) -> bool {
48        matches!(
49            self,
50            PieceType::WhitePawn
51                | PieceType::WhiteRook
52                | PieceType::WhiteKnight
53                | PieceType::WhiteBishop
54                | PieceType::WhiteQueen
55                | PieceType::WhiteKing
56        )
57    }
58
59    pub fn from_string(s: &char) -> Option<PieceType> {
60        match s {
61            'P' => Some(PieceType::WhitePawn),
62            'R' => Some(PieceType::WhiteRook),
63            'N' => Some(PieceType::WhiteKnight),
64            'B' => Some(PieceType::WhiteBishop),
65            'Q' => Some(PieceType::WhiteQueen),
66            'K' => Some(PieceType::WhiteKing),
67            'p' => Some(PieceType::BlackPawn),
68            'r' => Some(PieceType::BlackRook),
69            'n' => Some(PieceType::BlackKnight),
70            'b' => Some(PieceType::BlackBishop),
71            'q' => Some(PieceType::BlackQueen),
72            'k' => Some(PieceType::BlackKing),
73            _ => None,
74        }
75    }
76    pub fn to_char(self) -> char {
77        match self {
78            PieceType::WhitePawn => 'P',
79            PieceType::WhiteRook => 'R',
80            PieceType::WhiteKnight => 'N',
81            PieceType::WhiteBishop => 'B',
82            PieceType::WhiteQueen => 'Q',
83            PieceType::WhiteKing => 'K',
84            PieceType::BlackPawn => 'p',
85            PieceType::BlackRook => 'r',
86            PieceType::BlackKnight => 'n',
87            PieceType::BlackBishop => 'b',
88            PieceType::BlackQueen => 'q',
89            PieceType::BlackKing => 'k',
90            _ => panic!(),
91        }
92    }
93}

95impl Not for PieceType {
96    type Output = Self;
97
98    fn not(self) -> Self::Output {
99        match self {
100            PieceType::WhitePawn => PieceType::BlackPawn,
101            PieceType::WhiteRook => PieceType::BlackRook,
102            PieceType::WhiteKnight => PieceType::BlackKnight,
103            PieceType::WhiteBishop => PieceType::BlackBishop,
104            PieceType::WhiteQueen => PieceType::BlackQueen,
105            PieceType::WhiteKing => PieceType::BlackKing,
106            PieceType::BlackPawn => PieceType::WhitePawn,
107            PieceType::BlackRook => PieceType::WhiteRook,
108            PieceType::BlackKnight => PieceType::WhiteKnight,
109            PieceType::BlackBishop => PieceType::WhiteBishop,
110            PieceType::BlackQueen => PieceType::WhiteQueen,
111            PieceType::BlackKing => PieceType::WhiteKing,
112            _ => self,
113        }
114    }
115}
