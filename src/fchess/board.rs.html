31pub fn print_board(pieces: Vec<Piece>, f: &mut fmt::Formatter<'_>) -> fmt::Result {
32    let coords: Vec<Square> = pieces
33        .clone()
34        .into_iter()
35        .map(|piece| (piece.get_square()))
36        .collect();
37    writeln!(f, "  ┌───┬───┬───┬───┬───┬───┬───┬───┐")?;
38    for i in 0..8 {
39        write!(f, "{} ", 8 - i)?;
40        for j in 0..8 {
41            if coords.contains(&Square::from_rank_file(7 - i, j)) {
42                let index = coords
43                    .iter()
44                    .position(|r| *r == Square::from_rank_file(7 - i, j))
45                    .unwrap();
46                write!(f, "│ {} ", pieces[index].get_type())?;
47            } else {
48                write!(f, "│   ")?;
49            }
50        }
51
52        if i != 7 {
53            write!(f, "│\n  ├───┼───┼───┼───┼───┼───┼───┼───┤\n")?;
54        }
55    }
56    write!(f, "│\n  └───┴───┴───┴───┴───┴───┴───┴───┘\n")?;
57    writeln!(f, "    a   b   c   d   e   f   g   h  ")?;
58    f.write_str("")
59}

88    fn next(&mut self) -> Option<Piece> {
89        while (self.index as u64) < 64_u64 * 13_u64 {
90            let pieces_index: u8 = (self.index / 64) as u8;
91            let board_index: u8 = (self.index % 64) as u8;
92
93            self.index += 1;
94            if (self.board.pieces[pieces_index as usize] >> board_index) & 1 == 1 {
95                return Some(Piece::new(
96                    Square::from_index(board_index),
97                    self.board.piece_at(Square::from_index(board_index))?,
98                ));
99            }
100        }
101
102        None
103    }

159    pub fn from_fen(fen: &str) -> Board {
160        let vec = ['P', 'R', 'N', 'B', 'Q', 'K', 'p', 'r', 'n', 'b', 'q', 'k'];
161
162        let mut board = Board {
163            ..Default::default()
164        };
165
166        let mut rank: u8 = 0;
167        let mut file: u8 = 0;
168
169        let mut chars = fen.chars().collect::<VecDeque<char>>();
170        loop {
171            if chars.is_empty() {
172                break;
173            }
174
175            let c = chars.pop_front().unwrap();
176            let pos = vec.iter().position(|&r| r == c);
177
178            match (pos, c) {
179                (Some(o), _) => {
180                    let piece: PieceType = num::FromPrimitive::from_usize(o).unwrap();
181                    board.set_piece(Square::from_rank_file(7 - file, rank), piece);
182                    rank += 1;
183                }
184                (_, '/') => {
185                    rank = 0;
186                    file += 1;
187                }
188                (_, ' ') => break,
189                (_, '0'..='9') => rank += c.to_digit(10).unwrap() as u8,
190                _ => continue,
191            }
192        }
193
194        let tail_re =
195            Regex::new(r"([wb])? ?(K?)(Q?)(k?)(q?)-? ?(-?([a-h][1-8])?) ?(\d{1,2})? ?(\d{1,2})?")
196                .unwrap();
197
198        let tail = chars.iter().collect::<String>();
199        let captures = tail_re.captures(&tail).unwrap();
200
201        board.set_turn(match captures.get(1).map(|side| side.as_str()) {
202            None | Some("w") => Side::White,
203            Some("b") => Side::Black,
204            _ => Side::White,
205        });
206
207        board.set_castling_white_short(captures.get(2).is_some_and(|key| key.as_str() == "K"));
208        board.set_castling_white_long(captures.get(3).is_some_and(|key| key.as_str() == "Q"));
209        board.set_castling_black_short(captures.get(4).is_some_and(|key| key.as_str() == "k"));
210        board.set_castling_black_long(captures.get(5).is_some_and(|key| key.as_str() == "q"));
211
212        board.set_enpassant(
213            captures
214                .get(7)
215                .and_then(|key| Square::from_algebraic(key.as_str())),
216        );
217
218        board.set_half_move_clock(
219            captures
220                .get(8)
221                .map_or(0, |key| key.as_str().parse::<u8>().unwrap()),
222        );
223
224        board.set_full_move_clock(
225            captures
226                .get(9)
227                .map_or(0, |key| key.as_str().parse::<u8>().unwrap()),
228        );
229
230        board
231    }

266    fn is_castle(&self, mov: Move) -> Option<(Move, Move, Castling)> {
267        let piece_type = self.piece_at(mov.get_src())?;
268        let src_rank = mov.get_src().get_rank();
269        let src_file = mov.get_src().get_file();
270        let dst_rank = mov.get_dst().get_rank();
271        let dst_file = mov.get_dst().get_file();
272
273        match ((src_rank, src_file), (dst_rank, dst_file), piece_type) {
274            ((0, 4), (0, 2), PieceType::WhiteKing) => Some((
275                mov,
276                Move::new(Square::from_rank_file(0, 0), Square::from_rank_file(0, 3)),
277                Castling::WhiteLong,
278            )),
279            ((0, 4), (0, 6), PieceType::WhiteKing) => Some((
280                mov,
281                Move::new(Square::from_rank_file(0, 7), Square::from_rank_file(0, 5)),
282                Castling::WhiteShort,
283            )),
284            ((7, 4), (7, 2), PieceType::BlackKing) => Some((
285                mov,
286                Move::new(Square::from_rank_file(7, 0), Square::from_rank_file(7, 3)),
287                Castling::BlackLong,
288            )),
289            ((7, 4), (7, 6), PieceType::BlackKing) => Some((
290                mov,
291                Move::new(Square::from_rank_file(7, 7), Square::from_rank_file(7, 5)),
292                Castling::BlackShort,
293            )),
294            _ => None,

298    fn apply_single_move(self: &Board, mov: Move) -> Option<Board> {
299        let mut result = self.clone();
300
301        let piece_type = self.piece_at(mov.get_src()).unwrap() as usize;
302
303        for i in Scope::All.to_range() {
304            result.pieces[i] &= 0xFFFFFFFFFFFFFFFF ^ (1 << mov.get_dst().get_index());
305        }
306
307        result.pieces[piece_type] &= 0xFFFFFFFFFFFFFFFF ^ (1 << mov.get_src().get_index());
308
309        result.pieces[piece_type] |= 1 << mov.get_dst().get_index();
310
311        Some(result)
312    }

314    pub fn apply(self: &Board, mov: Move) -> Option<Board> {
315        let castle = self.is_castle(mov.clone());
316        let mut result = if castle.is_some() {
317            let castle = castle.unwrap();
318            let mut board = self.apply_single_move(castle.0)?;
319            match castle.2 {
320                Castling::WhiteShort => board.set_castling_white_short(false),
321                Castling::WhiteLong => board.set_castling_white_long(false),
322                Castling::BlackShort => board.set_castling_black_short(false),
323                Castling::BlackLong => board.set_castling_black_long(false),
324            }
325            board.apply_single_move(castle.1)
326        } else {
327            self.apply_single_move(mov.clone())
328        }?;
329
330        if mov.get_promotion().is_some() {
331            let piece_type = self.piece_at(mov.get_src()).unwrap() as usize;
332            result.pieces[piece_type] &= 0xFFFFFFFFFFFFFFFF ^ (1 << mov.get_dst().get_index());
333            result.pieces[mov.get_promotion().unwrap() as usize] |= 1 << mov.get_dst().get_index();
334        }
335
336        match (mov.get_src().get_rank(), mov.get_src().get_file()) {
337            (0, 0) => result.set_castling_white_long(false),
338            (0, 7) => result.set_castling_white_short(false),
339            (7, 0) => result.set_castling_black_long(false),
340            (7, 7) => result.set_castling_black_short(false),
341            (0, 4) => {
342                result.set_castling_white_long(false);
343                result.set_castling_white_short(false);
344            }
345            (7, 4) => {
346                result.set_castling_black_long(false);
347                result.set_castling_black_short(false);
348            }
349            _ => {}
350        }
351        result.set_enpassant(None);
352
353        let piece_type = self.piece_at(mov.get_src()).unwrap();
354
355        if piece_type == PieceType::WhitePawn {
356            if mov.get_dst().get_rank() - mov.get_src().get_rank() == 2 {
357                result.set_enpassant(Some(Square::from_rank_file(
358                    mov.get_src().get_rank() + 1,
359                    mov.get_src().get_file(),
360                )));
361            }
362        } else if piece_type == PieceType::BlackPawn
363            && mov.get_src().get_rank() - mov.get_dst().get_rank() == 2
364        {
365            result.set_enpassant(Some(Square::from_rank_file(
366                mov.get_dst().get_rank() + 1,
367                mov.get_dst().get_file(),
368            )));
369        }
370
371        if Some(mov.get_dst()) == self.get_enpassant() {
372            let clear_square = if piece_type == PieceType::BlackPawn {
373                Some(Square::from_rank_file(
374                    mov.get_dst().get_rank() + 1,
375                    mov.get_dst().get_file(),
376                ))
377            } else if piece_type == PieceType::WhitePawn {
378                Some(Square::from_rank_file(
379                    mov.get_dst().get_rank() - 1,
380                    mov.get_dst().get_file(),
381                ))
382            } else {
383                None
384            };
385
386            if let Some(clear_square) = clear_square {
387                result.pieces[!piece_type as usize] &= !(1 << clear_square.get_index());
388            }
389        }
390
391        result.set_turn(!self.get_turn());
392        if result.get_turn() == Side::White {
393            result.set_full_move_clock(result.get_full_move_clock() + 1);
394        }
395
396        let moved_piece = self.piece_at(mov.get_src()).unwrap();
397        let target_piece = self.piece_at(mov.get_dst()).unwrap();
398
399        let mov = (moved_piece, target_piece);
400
401        let halfmove_clock_reset = matches!(mov, (PieceType::WhitePawn, _))
402            | matches!(mov, (PieceType::BlackPawn, _))
403            | matches!(mov, (_, PieceType::WhitePawn))
404            | matches!(mov, (_, PieceType::BlackPawn))
405            | (target_piece != PieceType::NoPiece);
406
407        if halfmove_clock_reset {
408            result.set_half_move_clock(0);
409        } else {
410            result.set_half_move_clock(self.get_half_move_clock() + 1);
411        }
412
413        if self.get_enpassant().is_some() {
414            result.set_enpassant(None);
415        }
416
417        Some(result)
418    }
